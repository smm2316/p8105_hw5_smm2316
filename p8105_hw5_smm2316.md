``` r
clean_iris = function(x) {
  
   if (is.numeric(x)) {
   replace_na(x, mean(x, na.rm = TRUE))
   }
   else if (is.character(x)) 
     {replace_na(x, "virginica")}
}

output = map_dfr(.x = iris_with_missing, ~ clean_iris(.x))
output %>%
  view()
```
